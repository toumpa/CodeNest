Βήμα 8: Υλοποίηση ενεργειών μενού - Υπολογισμός Μικρότερου Στοιχείου

Όπως αρχικοποιήσαμε μια μεταβλητή για το άθροισμα τώρα θα προσθέσουμε μια για να αποθηκεύσουμε το μικρότερο στοιχείο και στην συνέχεια να το εμφανίσουμε.

Το στοιχείο αυτό όμως, με τι θα χρειαστεί να το ορίσουμε αρχικά ίσο;
Για να το συγκρίνουμε λοιπόν με όλα τα στοιχεία του πίνακα και να αποφασίσουμε ποιό είναι το μικρότερο χρειάζεται αυτή η μεταβλητή να έχει μια αρχική τιμή η οποία θα είναι 100% μεγαλύτερη από όλα τα στοιχεία του πίνακα για να μπορεί να άλλαξει στην πορεία με την τιμή του στοιχείου του πίνακα με την μικρότερη τιμή.

Στην άσκηση νας γνωρίζουμε ότι όλες οι τιμές που υπάρχουν στον πίνακα είναι από 1 έως 100 οπότε αυτόματα σκεφτόμαστε πως η μικρότερη τιμή είναι μια χαμηλότερη του 1 και αντίστοιχα μια μεγαλύτερη από όλες αυτές που έδωσε ο χρήστης είναι μια υψηλότερη του 100.
Έστω, λοιπόν, η μεταβλητή min που θα μπορούσε να έχει αρχική τιμή 0 (< 1).

*Στα πλαίσια του διαδικτυακού μας μαθήματος θα δούμε και την γενική περίπτωση που δεν γνωρίζουμε εξ'αρχής το εύρος τιμών ενός πίνακα.
Αυτό το πρόβλημα λύνεται πολύ απλά αρχικοποιώντας την μεταβλητή min με μια τιμή που ήδη υπάρχει μέσα στον πίνακα, έστω την πρώτη (Α[0]).
Κατά αυτόν τον τρόπο συγκρίνει το πρόγραμμα την τιμή του πρώτου στοιχείου με όλες τις υπόλοιπες του πίνακα και αν βρεθεί κάποια μικρότερη του 1ου στοιχείου τότε αυτή αλλάζει τιιμή. 

int min; //to orizoume mazi me ta upoloipa

Βήμα 10: Υλοποίηση ενεργειών μενού - Αναζήτηση Συγκεκριμένου Στοιχείου

Το πρώτο πράγμα που χρειαζόμαστε για αυτήν την ενέργεια είναι ένα στοιχείο αναζήτησης. Ζητάμε από τον χρήστη να εισάγει μια τιμή και την αποθηκεύουμε σε μια μεταβλητή, έστω key.
Επίσης θα χρειαστούμε μια ακόμη μετάβλητη flag η οποία θα αποσαφηνίζει εαν έχει βρεθεί η τιμή αναζήτησης στον πίνακα ή όχι.

Με την γνωστή σε όλους μας πλέον δομή for προσπελαύνει το πρόγραμμα όλα τα δεδομένα του πίνακα και συγκρίνει αν η τιμή που θέλουμε να βρούμε ταιριάζει με κάποια από αυτές του πίνακα.
Αν βρεθεί αντιστοιχία τότε αλλάζουμε την τιμή της flag από 0 που την ορίσαμε σε 1 και επιπλέον μπορούμε να κρατήσουμε και την θέση (spot) στην οποία βρέθηκε η συγκεκριμένη τιμή στον πίνακα για να την εμφανίσουμε στην οθόνη.

Τέλος, εκτυπώνουμε στην οθόνη το κατάλληλο μύνημα αναλόγως με το αν βρέθηκε η τιμή αναζήτησης ή όχι. Πολύ εύκολα τσεκάρουμε εαν η τιμή της flag έχει αλλάξει σε 1.

Πάμε να το δούμε ολοκληρωμένο: 

int key, flag, spot;
else if (choice == 4) {

    printf("Dwse to stoixeio pou psaxneis:\t");
    scanf("%d", &key);      // key einai to stoixeio anazhthshs

    //orizoume  metablhth an brethike h oxi to stoixeio
    flag = 0;   //an parameinei 0 tote to stoixeio den brethike
    spot = -1;
    //bazoume -1 oxi 0 giati oi theseis tou pinaka arxizoun apo 0

    for(i = 0; i < size; i++){
        if(key == A[i]){
            flag = 1;   //brethike to stoixeio allazw thn flag se 1
            spot = i;   //kratame thn thesh pou brethike
        }
    }

    if(flag == 1){  //brethike
        printf("\nBrethike to %d sthn thesh -> %d\n", key, spot+1);
        //+1 giati arxizoun apo to 0 oi theseis   
    }
    else{   // den brethike
        printf("\nTo stoixeio den brethike ston pinaka!\n");
    }
}

Βήμα 11: Υλοποίηση ενεργειών μενού - Διαγραφή Συγκεκριμένου Στοιχείου


Αυτή η ενέργεια είναι η πιο περίπλοκη από όλες αλλά θα την δούμε σιγά σιγά παρέα.

Ποιό ήταν το πρώτο βήμα που κάναμε πριν στην αναζήτηση; Να αποθηκεύσουμε το στοιχείο που αναζητούμε!
Το ίδιο θα γίνει και εδώ. Για να διαγράψεις ένα στοιχείο πρέπει πρώτα να το βρεις. Σωστά;

Ζητάμε από τον χρήστη να μας δώσει την τιμή που θέλει να διαγράψει από τον πίνακα (αν αυτή υπάρχει). Μπορούμε να ορίσουμε μια νέα μεταβλητή, αλλά στο συγκεκριμένο πρόγραμμα δεν είναι απαραίτητο καθώς μπορούμε να χρησιμοποιήσουμε την ποηγούμενη μεταβλητή key.
Επιπλέον, πάλι θα ορίσουμε μια νέα μεταβλητή found με την ίδια λογική της μεταβλητής flag.

Θα χρειαστεί όπως καταλαβαίνεις να δημιουργηθεί μια επανάληψη για να βρεθεί το στοιχείο διαγραφής. Θα χρησιμοποιήσουμε την while.
Πρέπει οποσδήποτε να γίνει έλεγχος αν υπάρχουν στοιχεία μέσα στον πίνακα, για την περίπτωση που ο χρήστης τα έχει διαγράψει όλα. Αυτός είναι και ο λόγος που ορίσαμε στην αρχή του προγράμματος την μεταβλητή size=10 για έχουμε την δυνατότητα να "επεξεργαστούμε" το μέγεθος του πίνακα και παράλληλα να μπορούν να λειτουργούν κανονικά όλες οι υπόλοιπες ενέργειες του μενού.

Αν βρούμε δηλαδή το στοιχείο διαγραφής και υπάρχουν στοιχεία στον πίνακα, τότε μεταφέρουμε όλα τα στοιχεία του πίνακα που υπάρχουν ΜΕΤΑ από αυτό, μια θέση προς τα αριστερά!
Αυτό θα έχει ως αποτέλεσμα το στοιχείο διαγραφής να μετακινηθεί στην τελευταία θέση του πίνακα και εφόσον εγω μειώσω το μέγεθος του αυτό να διαγραφτεί.
Η διαδικασία μετακίνησης ονομάζεται ολίσθηση.

Σε κάθε περίπτωση (είτε βρεθεί, είτε όχι) πρέπει ο δείκτης i του πίνακα να αυξάνεται για να μπορεί το πρόγραμμα να προσπελάσει όλα τα στοιχεία του.

Τέλος, όπως και στην αναζήτηση, έτσι και στην διαγραφή ελέγχουμε με την βοήθεια της found αν διαγράφθηκε το στοιχείο key ή όχι και να εμφανίσουμε το αντίστοιχο μύνημα.
Αν η διαγραφή ήταν επιτυχής, στην οθόνη θα εκτυπωθεί και ο νέος πίνακας χωρίς το στοιχείο διαγραφής. 

int found, temp; //prosthetoume sthn arxh
else if (choice == 5) {
    printf("\nDwse to stoixeio pou thes na diagrapseis:\t");
    scanf("%d", &key);      

    i = 0;
    found = 0;  //an brethei to stoixeio diagrafhs

    //an den exei brethei to stoixeio KAI yparxoun stoixeia
    while(i < size && found == 0){    
        if (key == A[i]){   //anazhthsh stoixeiou
            found = 1;      //brethike

            //olisthisi mia thesh pisw apo thn thesh pou brethike to key
            temp = A[i];    
            A[i] = A[size-1];
            A[size-1] = temp;

            size--; //meiwnw to megethos tou pinaka
        }
        i++;  //phgainw sto epomeno stoixeio   
    }

    if(found == 1){     //brethike
        printf("\nO arithmos %d entopisthke kai diagrafthke.\n", key);
        printf("Neos Pinakas: \n");

        for(i = 0; i < size ; i++){
            printf ("%d\t", A[i]);
        }
        printf ("\n");
    }
    else{
        printf("\nO aritmos pou edwses den uparxei\n");
    }
}

            printf("Epileje mia energeia: 0-5\n");
            scanf("%d", &choice);   //diavazw thn epilogh tou menu
    
    
            if (choice == 0) {      //an dwsei 0 termatizei to programma
                break;
            }
            else if (choice == 1) {
    
                sum = 0;
                for(i = 0; i < size; i++) {
                    sum += A[i];
                }
                printf("\nTo athroisma twn stoixeiwn einai: %d\n", sum);
            }
            else if (choice == 2) {
                
                min = A[0];
                for(i = 1; i < size; i++) {
                    if (A[i] < min) {
                        min = A[i];
                    }
                }
                printf("\nTo mikrotero stoixeio einai: %d\n", min);
            }
            else if (choice == 3) {
    
                max = A[0];
                for(i = 1; i < size; i++) {
                    if (A[i] > max) {
                        max = A[i];
                    }
                }
                printf("\nTo megalytero stoixeio einai: %d\n", max);
            }
            else if (choice == 4) {
    
                printf("Dwse to stoixeio pou psaxneis:\t");
                scanf("%d", &key);      // key einai to stoixeio anazhthshs
    
                // orizoume mia metablhth pou leei an brethike h oxi to stoixeio anazhthshs
                flag = 0;     // an parameinei 0 tote to stoixeio den brethike
                spot = -1;     // bazoume -1 oxi 0 giati oi theseis tou pinaka arxizoun apo 0
    
                for(i = 0; i < size; i++){
                    if(key == A[i]){
                        flag = 1;   // brethike to stoixeio allazw thn flag se 1
                        spot = i;   // kratame thn thesh pou brethike
                    }
                }
    
                if(flag == 1){  //an brethike
                    printf("\nBrethike to stoixeio %d, sthn thesh -> %d\n", key, spot+1);
                    //spot+1 giati arxizoun apo to 0 oi theseis   
                }
                else{
                    printf("\nTo stoixeio den brethike ston pinaka!\n");
                }
            }
            else if (choice == 5) {
                printf("\nDwse to stoixeio pou thes na diagrapseis:\t");
                scanf("%d", &key);      // key einai to stoixeio diagrafhs
    
                i = 0;
                found = 0;  // an brethei to stoixeio diagrafhs
    
                // an den exei brethei to stoixeio akoma KAI yparxoun stoixeia mesa ston pinaka
                while(i < size && found == 0){    
                    if (key == A[i]){   // anazhthsh stoixeiou
                        found = 1;      // brethike
    
                        //metaferw ola ta stoixeia tou pinaka mia thesh pisw
                        //apo thn thesh pou brethike to stoixeio diagrafhs
                        temp = A[i];    
                        A[i] = A[size-1];
                        A[size-1] = temp;
    
                        size--; // meiwnw to megethos tou pinaka
                    }
                    i++;  // phgainw sto epomeno stoixeio 
                    // DEN kanw break gia na elegjw an uparxei kai allh fora to stoixeio ston pinaka  
                }
    
                if(found == 1){     // an brethike
                    printf("\nO arithmos %d entopisthke kai diagrafthke.\n", key);
                    printf("Neos Pinakas: \n");
    
                    for(i = 0; i < size ; i++){
                        printf ("%d\t", A[i]);
                    }
                    printf ("\n");
                }
                else{
                    printf("\nO aritmos pou edwses den uparxei\n");
                }
            }
            else {
                printf("Mh egkiri epilogh! Dwse arithmo 0-5\n");
            }
    
        }
    
        return 0;
