Ας γνωρίσουμε την γλώσσα C

Προγραμματισμός ενός προβλήματος σε γλώσσα C βασισμένο στις ενότητες των μαθημάτων της πλατφόρμας


Θέμα:

Διαχείριση Βιβλίων μιας Βιβλιοθήκης

Σκοπός:

Δημιουργία ενός προγράμματος σε C που διαχειρίζεται τα βιβλία μιας βιβλιοθήκης. Το πρόγραμμα θα πρέπει να επιτρέπει την καταχώριση, την προβολή, την αναζήτηση, τη διαγραφή, την ταξινόμηση των βιβλίων κατά τίτλο ή έτος, και την παρακολούθηση δανεισμού και επιστροφής βιβλίων.

Απαιτήσεις:


Δομή Βιβλίου (Book): Χρησιμοποιήστε μια δομή (struct) για να αποθηκεύσετε τις πληροφορίες ενός βιβλίου. Κάθε βιβλίο θα έχει:
📍Τίτλο (συμβολοσειρά)
📍Συγγραφέα (συμβολοσειρά)
📍Έτος έκδοσης (ακέραιος αριθμός)
📍Κατάσταση Δανεισμού (ακέραιος αριθμός για να δείξει αν το βιβλίο είναι διαθέσιμο ή δανεισμένο)
📍Πίνακας Βιβλίων: Χρησιμοποιήστε έναν στατικό πίνακα για να αποθηκεύσετε μέχρι 100 βιβλία.

Συναρτήσεις:
📍addBook: Προσθέτει ένα νέο βιβλίο στη βιβλιοθήκη.
📍displayBooks: Εμφανίζει όλα τα βιβλία στη βιβλιοθήκη.
📍searchBookByTitle: Αναζητά ένα βιβλίο με βάση τον τίτλο του και εμφανίζει τις πληροφορίες του αν βρεθεί.
📍deleteBook: Διαγράφει ένα βιβλίο από τη βιβλιοθήκη με βάση τον τίτλο.
📍borrowBook: Επιτρέπει τη δανειστική χρήση ενός βιβλίου αν είναι διαθέσιμο.
📍returnBook: Επιστρέφει ένα βιβλίο αν είχε δανειστεί.

Δομές Επαναλήψης και Επιλογής:
Χρησιμοποιήστε επαναληπτικές δομές (for, while) για να περιηγηθείτε και να διαχειριστείτε τα βιβλία.
Χρησιμοποιήστε δομές επιλογής (if) για να ελέγξετε τις συνθήκες (π.χ. αν υπάρχει ένα βιβλίο με τον δοθέντα τίτλο ή αν είναι διαθέσιμο για δανεισμό).

Δείκτες: Χρησιμοποιήστε δείκτες για να περάσετε παραμέτρους στις συναρτήσεις.

Ταξινόμηση Βιβλίων: Ενσωματώστε δυνατότητες ταξινόμησης των βιβλίων κατά:
📍Τίτλο (Αύξουσα ή Φθίνουσα σειρά)
📍Έτος έκδοσης (Αύξουσα ή Φθίνουσα σειρά)

Επιπλέον Οδηγίες:

💠Χρησιμοποιήστε συναρτήσεις για την ταξινόμηση των βιβλίων κατά τίτλο και κατά έτος. Οι συναρτήσεις θα πρέπει να οργανώνουν τα βιβλία είτε αλφαβητικά είτε χρονολογικά.
💠Εξασφαλίστε ότι η ταξινόμηση μπορεί να γίνει είτε σε αύξουσα είτε σε φθίνουσα σειρά.

💠Το πρόγραμμα θα πρέπει να συνεχίζει να εκτελείται μέχρι ο χρήστης να επιλέξει την έξοδο.

#include <stdio.h>
#include <string.h>

typedef struct {
    char title[100];
    char author[100];
    int year;
    int isBorrowed; // 0 an to vivlio einai diathesimo, 1 an exei daneistei
} Book;

// Pinakas vivlion
Book library[100];
int bookCount = 0; // Arithmos vivlion sti vivliothiki

void sortByTitleAsc(Book *library, int bookCount) {
    Book temp;
    for (int i = 0; i < bookCount - 1; i++) {
        for (int j = i + 1; j < bookCount; j++) {
            if (strcmp(library[i].title, library[j].title) > 0) {
                temp = library[i];
                library[i] = library[j];
                library[j] = temp;
            }
        }
    }
}

void sortByTitleDesc(Book *library, int bookCount) {
    Book temp;
    for (int i = 0; i < bookCount - 1; i++) {
        for (int j = i + 1; j < bookCount; j++) {
            if (strcmp(library[i].title, library[j].title) < 0) {
                temp = library[i];
                library[i] = library[j];
                library[j] = temp;
            }
        }
    }
}

void sortByYearAsc(Book *library, int bookCount) {
    Book temp;
    for (int i = 0; i < bookCount - 1; i++) {
        for (int j = i + 1; j < bookCount; j++) {
            if (library[i].year > library[j].year) {
                temp = library[i];
                library[i] = library[j];
                library[j] = temp;
            }
        }
    }
}

void sortByYearDesc(Book *library, int bookCount) {
    Book temp;
    for (int i = 0; i < bookCount - 1; i++) {
        for (int j = i + 1; j < bookCount; j++) {
            if (library[i].year < library[j].year) {
                temp = library[i];
                library[i] = library[j];
                library[j] = temp;
            }
        }
    }
}

    printf("2. Kata titlo (Fthinousa)\n");

    printf("4. Kata etos (Fthinousa)\n");
    printf("Epilogi: ");
    scanf("%d", &option);
    getchar(); // Katharismos tou xaraktira newline
    
    while (option < 1 || option > 4) {
        printf("Lathos epilogi. Parakalw epilxte 1, 2, 3 i 4: ");
        scanf("%d", &option);
        getchar(); // Katharismos tou xaraktira newline
    }
    return option;
}

// Synartisi Taxinomisis Vivlion
void sortBooks(Book *library, int bookCount) {
    int option = getSortOption();

    if (option == 1) {
        sortByTitleAsc(library, bookCount);
        printf("Taxinomisi kata titlo (auksousa) oloklhrwthike.\n");
    } else if (option == 2) {
        sortByTitleDesc(library, bookCount);
        printf("Taxinomisi kata titlo (fthinousa) oloklhrwthike.\n");
    } else if (option == 3) {
        sortByYearAsc(library, bookCount);
        printf("Taxinomisi kata etos (auksousa) oloklhrwthike.\n");
    } else if (option == 4) {
        sortByYearDesc(library, bookCount);
        printf("Taxinomisi kata etos (fthinousa) oloklhrwthike.\n");
    }
}

// Synartisi gia sygkrisi titlon vivlion (xrisimopoiitai gia taxinomisi)
int compareBooks(const void *a, const void *b) {
    return strcmp(((Book *)a)->title, ((Book *)b)->title);
}

// Synartiseis
void addBook(Book *library, int *bookCount) {
    if (*bookCount >= 100) {
        printf("H vivliothiki einai pliris!\n");
        return;
    }
    
    Book newBook;
    
    printf("Eisagete ton titlo tou vivliou: ");
    fgets(newBook.title, 100, stdin);
    newBook.title[strcspn(newBook.title, "\n")] = '\0'; // Afairesi tou \n apo to fgets
    
    printf("Eisagete ton siggrafea tou vivliou: ");
    fgets(newBook.author, 100, stdin);
    newBook.author[strcspn(newBook.author, "\n")] = '\0'; // Afairesi tou \n apo to fgets
    
    printf("Eisagete to etos ekdosis tou vivliou: ");
    scanf("%d", &newBook.year);
    getchar(); // Katharismos tou xaraktira newline
    
    newBook.isBorrowed = 0; // To vivlio einai diathesimo otan prostethetai
    
    library[*bookCount] = newBook;
    (*bookCount)++;
    
    printf("To vivlio prostethike epityxws!\n");
}

void displayBooks(Book *library, int bookCount) {
    if (bookCount == 0) {
        printf("H vivliothiki einai keni!\n");
        return;
    }
    
    // Taxinomisi vivlion alfabitika kata titlo
    // qsort(library, bookCount, sizeof(Book), compareBooks);
    
    printf("\nLista vivlion:\n");
    for (int i = 0; i < bookCount; i++) {
        printf("Titlos: %s\n", library[i].title);
        printf("Siggrafeas: %s\n", library[i].author);
        printf("Etos Ekdosis: %d\n", library[i].year);
        printf("Katastasi Daneismou: %s\n", library[i].isBorrowed ? "Daneismeno" : "Diathesimo");
        printf("-------------------------\n");
    }
}

void searchBookByTitle(Book *library, int bookCount, const char *title) {
    for (int i = 0; i < bookCount; i++) {
        if (strcmp(library[i].title, title) == 0) {
            printf("Vivlio vrethike!\n");
            printf("Titlos: %s\n", library[i].title);
            printf("Siggrafeas: %s\n", library[i].author);
            printf("Etos Ekdosis: %d\n", library[i].year);
            printf("Katastasi Daneismou: %s\n", library[i].isBorrowed ? "Daneismeno" : "Diathesimo");
            return;
        }
    }
    printf("Vivlio me titlo '%s' den vrethike sti vivliothiki.\n", title);
}

void deleteBook(Book *library, int *bookCount, const char *title) {
    for (int i = 0; i < *bookCount; i++) {
        if (strcmp(library[i].title, title) == 0) {
            for (int j = i; j < *bookCount - 1; j++) {
                library[j] = library[j + 1];
            }
            (*bookCount)--;
            printf("To vivlio me titlo '%s' diagraphike epityxws.\n", title);
            return;
        }
    }
    printf("Vivlio me titlo '%s' den vrethike gia diagrafi.\n", title);
}

void borrowBook(Book *library, int bookCount, const char *title) {
    for (int i = 0; i < bookCount; i++) {
        if (strcmp(library[i].title, title) == 0) {
            if (library[i].isBorrowed) {
                printf("To vivlio me titlo '%s' exei hdh daneistei.\n", title);
            } else {
                library[i].isBorrowed = 1;
                printf("To vivlio me titlo '%s' daneistike epityxws.\n", title);
            }
            return;
        }
    }
    printf("Vivlio me titlo '%s' den vrethike gia daneismo.\n", title);
}

void returnBook(Book *library, int bookCount, const char *title) {
    for (int i = 0; i < bookCount; i++) {
        if (strcmp(library[i].title, title) == 0) {
            if (library[i].isBorrowed) {
                library[i].isBorrowed = 0;
                printf("To vivlio me titlo '%s' epistrofike epityxws.\n", title);
            } else {
                printf("To vivlio me titlo '%s' den einai daneismeno.\n", title);
            }
            return;
        }
    }
    printf("Vivlio me titlo '%s' den vrethike gia epistrofi.\n", title);
}
